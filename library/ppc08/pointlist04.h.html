class PointListNode;
class PointList
{
public:
    PointList();
    ~PointList();
    PointList(const PointList& other) = delete;
    PointList& operator=(const PointList& rhs) = delete;
    PointList& operator+=(const Point& point);
    size_t size() const;
    const Point& front() const;
    void pop_front();

private:
    std::unique_ptr<PointListNode> mHead;
    PointListNode* mTail = nullptr;
    size_t mSize = 0;
};
