#include <iostream>
#include "point55.h"
int main()
{
    Point p;
    p.setX(-1.0).setY(-1.0).setColor("red"); // <> Verkettung der Mutatoren bei einer Variable `p`
    std::cout << p << std::endl;
    Point p2 = Point().setX(1.0).setY(2.0).setColor("red"); // <> Verkettung der Mutatoren in Kombination mit dem Kopierkonstruktor
    std::cout << p2 << std::endl;
    auto p3 = Point().setX(3.0).setY(4.0).setColor("cyan"); // <> Die Bedeutung des Parameterwertes ergibt sich bei _einstelligen_ Elementfunktionen direkt aus dem Namen der Funktion, deswegen darf hier der unspezifische Datentype `double` eingesetzt werden.
    std::cout << p3 << std::endl;
}
