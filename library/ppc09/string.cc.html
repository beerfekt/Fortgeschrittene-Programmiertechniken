#include "string.h"
void String::replace(std::string& s, const std::string& what, const std::string& with)
{
    const auto pos = s.find(what); // <> Der Gültigkeitsbereich von `pos` ist zu groß, er sollte auch den Körper des folgenden `if` eingeschränkt werden.
    if (pos != std::string::npos)
    {
        s.replace(pos, what.length(), with);
    }
}
