#if __has_include("../ppc.h")
# include "../ppc.h"
#endif
#include "svgfile.h"
#include "point80.h"
#include "line00.h"
int main()
{
    Point p1;
    SVGFile svgFile("test.svg");
    svgFile.add(p1); // <> Das `p1`-Objekt wird *per-reference* von `SVGFile::add(const SVGShape& shape)` entgegen genommen
    Line line1{Point{XCoor{1.0}, YCoor{1.0}}, Point{XCoor{100.0}, YCoor{100.0}}};
    line1.setColor("red");
    Line line2{{XCoor{1.0}, YCoor{1.0}}, {XCoor{100.0}, YCoor{-100.0}}};
    line2.setColor("blue");
    svgFile.add(line1); // <> Das `line1`-Objekt wird *per-reference* von `SVGFile::add(const SVGShape& shape)` entgegen genommen
    svgFile.add(line2); // <> wie oben
}
