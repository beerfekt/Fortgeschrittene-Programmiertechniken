#include <cstddef>
template<typename... T>
struct Tuple
{
    static constexpr size_t size()
    {
        return sizeof...(T);
    }
};
int main()
{
    Tuple<int, float> t1;
    static_assert(t1.size() == 2);

    auto t2 = Tuple<bool, int, double>();
    static_assert(t2.size() == 3);
}
