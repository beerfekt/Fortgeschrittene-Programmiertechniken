#include <iostream>
#include <cassert>
#include <cstdlib>
#include <utility> // <> Notwendig f√ºr den generischen Typ (template) `std::pair<T,U>`
int absoluteSum(const std::pair<int*, size_t>& array) // <> Generalisierung des Ansatzes mit POD-Typ (s.o.)
{
    assert(array.first); // <> Das Datenelement `first` hat hier den Typ `int*` und hat den Wert des Anfangszeigers.
    int sum = 0;
    for(size_t i = 0; i < array.second; ++i)   // <> Das Datenelement `second` ist hier vom Typ `size_t` und hat den Wert der Elementanzahl.
    {
        sum += std::abs(array.first[i]);
    }
    return sum;
}
int main()
{
    const size_t length = 5;
    int array[length] = {1, 2, -3, -4, 5};
    std::cout << absoluteSum({array, length}) << std::endl;
}
