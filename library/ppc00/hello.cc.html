#include "hello.h" // <> *Deklariert* die Klasse `Hello` (und die Funktion `Hello::main())
void Hello::main(std::vector<std::string> arguments)   // <> *Definition* der Element-Funktion `main()`
{

    for(std::string s: arguments)
    {
        std::cout << s << std::endl;
    }
}
