class Hello3
{
    public static void main(String[] arguments)   // <> Die Zeichenketten der Kommandozeilen Parameter werden als String-Array uebergeben
    {
        System.out.print("Hallo Java-Welt\n"); // <> Ausgabe eines String-Literals
        for(String s: arguments)   // <> Iteration ueber das gesamte Array `arguments`
        {
            System.out.println(s); // <> Ausgabe des jeweiligen Strings `s`
        }
    }
}
