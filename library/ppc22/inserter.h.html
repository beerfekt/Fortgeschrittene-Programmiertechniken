#pragma once
#if __has_include("../ppc.h")
# include "../ppc.h"
#endif
#include <iterator>
template<typename C>
class Inserter final : public std::iterator<std::output_iterator_tag, typename C::value_type>
{
public:
    Inserter(C& container) : mContainer(container) {}
    Inserter& operator*()
    {
        return *this;
    }
    Inserter& operator=(const typename C::value_type& v)
    {
        mContainer.push_back(v);
        return *this;
    }
    Inserter& operator++()
    {
        return *this;
    }
private:
    C& mContainer;
};
template<typename C>
Inserter<C> make_inserter(C& container)
{
    return Inserter<C>(container);
}
