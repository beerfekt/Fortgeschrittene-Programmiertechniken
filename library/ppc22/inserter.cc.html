#include "inserter.h"
#include "../lib/simpletest.h"
#include <vector>
SIMPLETEST("inserter1")
{
    std::vector<int> v1{1, 2, 3};
    std::vector<int> v2;
    std::copy(std::begin(v1), std::end(v1), Inserter<std::vector<int>>(v2));
    if (!std::equal(std::begin(v1), std::end(v1), std::begin(v2))) return false;
    return true;
};
SIMPLETEST("inserter2")
{
    std::vector<int> v1{1, 2, 3};
    std::vector<int> v2;
    std::copy(std::begin(v1), std::end(v1), make_inserter(v2));
    if (!std::equal(std::begin(v1), std::end(v1), std::begin(v2))) return false;
    return true;
};
