#include <iostream>
#include <string>
#include <algorithm>
#include "list13.h"
int main()
{
    List<std::string> l = {"eins", "zwei"};
    l += "drei";
    if (auto it2 = std::find(std::begin(l), std::end(l), "zwei"))   // <> Dies funktioniert nur mit einer expliziten Konversion zu `bool` und ist deswegen so nicht mit den Iteratoren der Container der {cppstdlib} machbar.
    {
        std::cout << *it2 << " found!" << std::endl;
    }
}
