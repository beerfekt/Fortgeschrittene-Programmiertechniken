#include <cstring>
#include <memory>
#define SIMULATION_STEPS 1000
#define NUM (1000UL*1000)
struct Dings_Pos
{
    Dings_Pos() : x(0), y(0) {}
    float x;
    float y;
    void move(float dx, float dy)
    {
        x += dx;
        y += dy;
    }
};
struct Dings_Name
{
    Dings_Name()
    {
        name[0] = '\0';
    }
    char name[512];
    void set_name(const char* n)
    {
        strcpy(name, n);
    }
};
int main()
{
    std::unique_ptr<Dings_Name[]>  data_name(new Dings_Name[NUM]);
    std::unique_ptr<Dings_Pos[]>  data_pos(new Dings_Pos[NUM]);
    for (size_t i = 0; i < NUM; ++i)
        data_name[i].set_name("default name");
    for (size_t st = 0; st < SIMULATION_STEPS; ++st)
    {
        for (size_t i = 0; i < NUM; ++i)
            data_pos[i].move(0.01, -0.02);
    }
    return 0;
}
