#include <iostream>
#include <type_traits>
template<typename T>
struct F
{
    F(const T& f) : mf{f}
    {
        std::cout << __PRETTY_FUNCTION__ << '\n';
    }
    void test()
    {
        std::cout << __PRETTY_FUNCTION__ << '\n';
        mf();
    }
    std::decay_t<T> mf;
//    const T& mf;
};

void g()
{
    std::cout << __PRETTY_FUNCTION__ << '\n';
}
int main()
{
    F<void(void)> f1(g);
    f1.test();

    auto g1 = g; // decay to function pointer

    F f2(g1);
    f2.test();

    F f3([]()  // lifetime?
    {
        std::cout << __PRETTY_FUNCTION__ << '\n';
    });
    f3.test();
    auto l1 = []()
    {
        std::cout << __PRETTY_FUNCTION__ << '\n';
    };
    F f4(l1);
    f4.test();
}
