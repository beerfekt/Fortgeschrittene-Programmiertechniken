#include <string>
#include <vector>
#include <utility>
class DingsContainer
{
    std::vector<std::string> names;
    std::vector<std::pair<float, float>> pos;
public:
    explicit DingsContainer(size_t size, const std::string& name = "default name") :
        names(size, name), pos(size)
    {
    }
    DingsContainer& moveAll(float x, float y)
    {
        for (auto& e: pos)
        {
            e.first  += x;
            e.second += y;
        }
        return *this;
    }
    DingsContainer& name(std::string&& nn)
    {
        for (auto& n: names)
        {
            n = std::move(nn);
        }
        return *this;
    }
};
int main()
{
    size_t SIMULATION_STEPS = 1000;
    size_t NUM = (1000UL*1000);
    DingsContainer data(NUM);
    for (size_t st = 0; st < SIMULATION_STEPS; ++st)
    {
        data.moveAll(1.0f, -2.0f);
    }
    return 0;
}
