#include <iostream>
#include <memory> // <> Header-Datei f√ºr `std::shared_ptr` und `std::make_shared`
#include "point61.h"
int main()
{
    std::shared_ptr<Point> sp1 = std::make_shared<Point>(XCoor{1.0}, YCoor{2.0}); // <> Um exlizite `new`-Operator-Aufrufe zu vermeiden, sollte das Funktionstemplate `std::make_shared<T>()` verwendet werden
    auto sp2 = std::make_shared<Point>(); // <> Der Typ von `sp2` wird aus dem Typ des Initialisierungsausdrucks abgeleitet
    std::shared_ptr<Point> sp3 = sp2; // <> Gemeinsame intelligente Zeiger (*SP*) sind *kopierbar*: `sp2` _und_ `sp3`  verwalten gemeinsam das Freispeicherobjekt `Point`
    std::cout << "Use count of sp1: " << sp1.use_count() << std::endl; // <> Wie viele gemeinsame Zeiger existieren?
    std::cout << "Use count of sp2: " << sp2.use_count() << std::endl;
    std::cout << "Use count of sp3: " << sp3.use_count() << std::endl;
    std::cout << *sp3 << std::endl; // <> Der Indirektionsoperator kann wie bei rohen Zeigern verwendet werden
}
