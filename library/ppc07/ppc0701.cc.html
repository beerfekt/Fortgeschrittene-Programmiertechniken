#include <iostream>
#include <vector> // <> Diese `include`-Datei ist für die Verwendung von `std::vector` notwendig.
#include "point60.h"
std::vector<Point> createPoints()
{
    std::vector<Point> points;
    Point p1{XCoor{0.0}, YCoor{0.0}};
    Point p2{XCoor{10.0}, YCoor{10.0}};
    Point p3{XCoor{20.0}, YCoor{20.0}};
    std::cout << p1 << std::endl;
    std::cout << p2 << std::endl;
    std::cout << p3 << std::endl;
    points.push_back(p1);
    points.push_back(p2);
    points.push_back(p3);
    return points;
}
int main()
{
    std::cout << "--- A ---" << std::endl;
    std::vector<Point> points = createPoints(); // <> Das Container-Template `std::vector` wird parametriert für den Elementtyp `Point`.
    std::cout << "--- B ---" << std::endl;
    for(const Point& p: points)
    {
        std::cout << p << std::endl;
    }
    std::cout << "--- C ---" << std::endl;
}
