#include <iostream>
#include <memory> // <> Header-Datei für `std::unique_ptr` und `std::make_unique`
#include <algorithm> // <> Header-Datei für `std::move`
#include "point61.h"
int main()
{
    std::unique_ptr<Point> up1 = std::make_unique<Point>(XCoor{1.0}, YCoor{2.0}); // <> `up1` ist ein *unikater* Zeiger (UP)
    auto up2 = std::make_unique<Point>(XCoor{3.0}, YCoor{4.0}); // <> Automatische Typinferenz
    // up2 = up1; // <> Kopierzuweisung nicht möglich
    up2 = std::move(up1); // <> Verschiebungszuweisung ist möglich, `up1` ist danach _leer_.
    std::cout << *up2 << std::endl; // <> Der Indirektionsoperator kann wie bei rohen Zeigern verwendet werden.
    if (up1)
    {
        std::cout << "Diese Ausgabe sollte nicht erscheinen ;-)" << std::endl;
    }
}
