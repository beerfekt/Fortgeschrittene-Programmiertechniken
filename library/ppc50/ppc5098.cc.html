#include <iostream>
template<typename T>
concept bool LessThanComparable()   //<> Definition eines `concept` `LessThanComparable` als _function concept_
{
    return requires (T a, T b)   //<> Anfang eines `requires`-Ausdrucks
    {
        a < b; // <> Ein einfaches `requirement`
    };
}
template<typename T>
requires LessThanComparable<T>() // <> Anwendung (`require`) eines _constraint_ `LessThanComparable` auf den Typ-Parametere `T` (hier: _function concept_)
const T& maximum(const T& a, const T& b)
{
    if (a < b)
    {
        return b;
    }
    return a;
}
const LessThanComparable& minimum(const LessThanComparable& a, const LessThanComparable& b)   //<> Platzhalter-Notation des _constraint_
{
    if (a < b)
    {
        return a;
    }
    return b;
}
